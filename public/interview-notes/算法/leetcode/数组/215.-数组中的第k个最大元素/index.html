    //用堆排序实现，删除前k-1个元素皆可以得到第k大的元素
    // 要求O(n),但是O(nlogn)可以通过

    //     }
    //    return nums[0];
        PriorityQueue<Integer> pq = new PriorityQueue<>((a,b)-> b-a);
        for (int i = 0; i < nums.length; i++) {
                pq.offer(nums[i]);
        }
        for (int i = 0; i < k - 1; i++) {
            pq.poll();
        } 
        
        int val = pq.poll();
        return val;
    }
